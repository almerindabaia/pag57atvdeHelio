package javaapplication7;
import javax.swing.JOptionPane;
        public class JavaApplication7 {
public static void main(String[] args) {
     
 String dia,mes;
 int dia1,mes1,dian;
        dia = JOptionPane.showInputDialog("Digite o dia ");
        dia1= Integer.parseInt(dia);
        mes= JOptionPane.showInputDialog("Digite o mês ");
        mes1= Integer.parseInt(mes);

 
        if (!(1 <= dia1 && dia1 <= 30) || !(1 <= mes1 && mes1 <= 12)) {
            JOptionPane.showMessageDialog(null, "Entrada inválida.");
        } else {
     
            dian = (mes1 - 1) * 30 + dia1;

            JOptionPane.showMessageDialog(null, "Desde o início passaram-se " + dian + " dias.");
        }
    }
}
